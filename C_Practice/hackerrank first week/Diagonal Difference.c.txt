#include <stdio.h>
#include <stdlib.h>

int main() {
    int n;
    scanf("%d", &n);
    int primary_sum = 0;
    int secondary_sum = 0;
    int val;

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &val);
            if (i == j) {
                primary_sum += val;
            }
            if (i + j == n - 1) {
                secondary_sum += val;
            }
        }
    }
    int dif = abs(primary_sum - secondary_sum);
    printf("%d\n", dif
    );

    return 0;
}
