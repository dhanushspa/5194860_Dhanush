#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int pos = 0, neg = 0, zero = 0;
    int arr[n];

    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
        if (arr[i] > 0) {
            pos++;
        } else if (arr[i] < 0) {
            neg++;
        } else {
            zero++;
        }
    }

    printf("%.6f\n", (double) pos / n);
    printf("%.6f\n", (double) neg / n);
    printf("%.6f\n", (double) zero / n);

    return 0;
}
