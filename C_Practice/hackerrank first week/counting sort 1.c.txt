#include <stdio.h>
int main() {
    int n;
    if (scanf("%d", &n) != 1) return 1;

    int freq[100] = {0};
    for (int i = 0; i < n; i++) {
        int x;
        scanf("%d", &x);
        if (x >= 0 && x < 100) {
            freq[x]++;
        }
    }

    for (int i = 0; i < 100; i++) {
        printf("%d", freq[i]);
        if (i < 99) putchar(' ');
    }
    putchar('\n');
    return 0;
}
