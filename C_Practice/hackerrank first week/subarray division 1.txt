#include <stdio.h>
#include <stdlib.h>

int solve(int n, int* s, int d, int m) {
    int i, j, sum, count = 0;
    for (i = 0; i <= n - m; i++) {
        for (j = i, sum = 0; j < i + m; j++) {
            sum += *(s + j);
        }
        if (sum == d) {
            count++;
        }
    }
    return count;
}
int main() {
    int n;
    scanf("%d", &n);
    int* s = malloc(n * sizeof(int));
    for (int i = 0; i < n; i++) {
        scanf("%d", &s[i]);
    }
    int d, m;
    scanf("%d %d", &d, &m);
    int result = solve(n, s, d, m);
    printf("%d\n", result);
    free(s);
    return 0;
}
